176 void XMLData::read(rapidxml::xml_node<>* node) {
177 // The node is the node of the data, so we can read everything from it right away, no navigation
178 // First, find the id
179 unsigned int readID = (unsigned int)std::stoi(node->first_attribute("id")->value());
180 m_attributes["id"]->setData(readID);
181 id = readID;
182 // Now, the name - returns a c_str, so we need to cast to std::string first
183 std::string readName = std::string(node->first_attribute("name")->value());
184 m_attributes["name"]->setData(readName);
185 name = readName;
186
187 // Now read every attribute - we read by reference so we can change the values that are stored in the class!
188 for(auto& attr: m_attributes) {
189 // Get some simple values
190 std::string attrName = attr.first;
191 AttributeBase* attrValue = attr.second;
192
193 // Make sure they're not ones we already loaded (name & id)
194 if(attrName == "name" || attrName == "id")
195 continue;
196
197 Logger::getInstance()->log("\t\tReading attribute " + attrName);
198
199 // Now, get the value based on the type of attribute
200 switch(attrValue->type) {
201 case AttributeType::STRING: {
202 std::string data = "";
203 // Use the templated function to get the string interpretation
204 XMLDataManager::readValue(node, attrName, data);
205 // Set the data now
206 attrValue->setData(data);
207 break;
208 }
209 case AttributeType::UNSIGNED_INT: {
210 unsigned int data = (unsigned int)-1;
211 // Use the templated function to get the string interpretation
212 XMLDataManager::readValue(node, attrName, data);
213 // Set the data now
214 attrValue->setData(data);
215 break;
216 }
217 case AttributeType::VECTOR_UNSIGNED_INT: {
218 std::vector<unsigned int> data = {};
219 // Use the templated function to get the string interpretation
220 XMLDataManager::readValue(node, attrName, data);
221 // Set the data now
222 attrValue->setData(data);
223 break;
224 }
225 case AttributeType::INT: {
226 int data = 0;
227 // Use the templated function to get the string interpretation
228 XMLDataManager::readValue(node, attrName, data);
229 // Set the data now
230 attrValue->setData(data);
231 break;
232 }
233 case AttributeType::FLOAT: {
234 float data = 0.0f;
235 // Use the templated function to get the string interpretation
236 XMLDataManager::readValue(node, attrName, data);
237 // Set the data now
238 attrValue->setData(data);
239 break;
240 }
241 case AttributeType::VECTOR_FLOAT: {
242 std::vector<float> data = {};
243 // Use the templated function to get the string interpretation
244 XMLDataManager::readValue(node, attrName, data);
245 // Set the data now
246 attrValue->setData(data);
247 break;
248 }
249 case AttributeType::BOOL: {
250 bool data = false;
251 // Use the templated function to get the string interpretation
252 XMLDataManager::readValue(node, attrName, data);
253 // Set the data now
254 attrValue->setData(data);
255 break;
256 }
257 case AttributeType::VECTOR_BOOL: {
258 std::vector<bool> data = {};
259 // Use the templated function to get the string interpretation
260 XMLDataManager::readValue(node, attrName, data);
261 // Set the data now
262 attrValue->setData(data);
263 break;
264 }
265 case AttributeType::VEC2: {
266 glm::vec2 data(0.0f);
267 // Use the templated function to get the string interpretation
268 XMLDataManager::readValue(node, attrName, data);
269 // Set the data now
270 attrValue->setData(data);
271 break;
272 }
273 case AttributeType::VECTOR_VEC2: {
274 std::vector<glm::vec2> data = {};
275 // Use the templated function to get the string interpretation
276 XMLDataManager::readValue(node, attrName, data);
277 // Set the data now
278 attrValue->setData(data);
279 break;
280 }
281 case AttributeType::SCRIPT: {
282 std::string path = "";
283 // Get the string interpretation
284 XMLDataManager::readValue(node, attrName, path);
285 // Load the script.
286 LuaScript scr = ResourceManager::loadScript(path);
287 // Set the data.
288 attrValue->setData(scr);
289 break;
290 }
291 case AttributeType::TEXTURE: {
292 std::string path = "";
293 // Get the path
294 XMLDataManager::readValue(node, attrName, path);
295
296 Texture tex;
297
298 if(path == "") {
299 // The data isn't available - default to an empty texture
300 } else {
301 // Just load it!
302 tex = ResourceManager::loadTexture(path);
303 }
304 // Set the data
305 attrValue->setData(tex);
306 break;
307 }
308
309 default: {
310 throwError(BAREError::XML_FAILURE,
311 "Failed to read attribute with type: " + std::to_string((unsigned int)attrValue->type) +
312 " of name " + attrName);
313 return;
314 }
315 };
316 }

319 void XMLData::write(rapidxml::xml_document<>* doc) {
320 // RapidXML requires that we allocate the name of the node by their function so it doesn't get cleaned up
321 char* allocatedNodeName = doc->allocate_string(nodeName.c_str());
322
323 // Now we can actually allocate the node, creating a new one for this data
324 rapidxml::xml_node<>* node = doc->allocate_node(rapidxml::node_element, allocatedNodeName);
325 doc->append_node(node);
326
327 // Write the name and id
328 {
329 // ID
330 // First, allocate the value as a c_string within rapidxml's systems
331 char* allocatedValue = doc->allocate_string(std::to_string(id).c_str());
332 // Now allocate the attribute itself as a node-ish thing
333 rapidxml::xml_attribute<>* idAttribute = doc->allocate_attribute("id", allocatedValue);
334 // Now actually add the attribute to the node
335 node->append_attribute(idAttribute);
336 }
337
338 {
339 // Name
340 // First, allocate the value as a c_string within rapidxml's systems
341 char* allocatedValue = doc->allocate_string(name.c_str());
342 // Now allocate the attribute itself as a node-ish thing
343 rapidxml::xml_attribute<>* nameAttribute = doc->allocate_attribute("name", allocatedValue);
344 // Now actually add the attribute to the node
345 node->append_attribute(nameAttribute);
346 }
347
348 // Now we're done with the simple name and id attributes, we need to add child nodes in the same way we loaded them
349 for(auto& attr: m_attributes) {
350 // Get some initial stuff
351 std::string attrName = attr.first;
352 AttributeBase* attrValue = attr.second;
353
354 // Make sure we aren't re-saving stuff we've already done
355 if(attrName == "name" || attrName == "id")
356 continue;
357
358 // We don't even really need to interpret (exterpret?) the value, as converting from data to a string is defined by the actual attribute, not by the type.
359
360 // Actually allocate that value, as well as the attribute's name
361 const char* allocatedValue = doc->allocate_string(attrValue->getDataString().c_str());
362 const char* allocatedName = doc->allocate_string(attrName.c_str());
363
364 // Allocate the node
365 rapidxml::xml_node<>* allocatedNode =
366 doc->allocate_node(rapidxml::node_element, allocatedName, allocatedValue);
367
368 // Actually add the node
369 node->append_node(allocatedNode);
370 }
371
372 // Now we've written all of the XML data to the document!
373 }
