181  for(unsigned int i = 0; i < m_numTextures; i++) {
182  // first, set the active texture to that of GL_TEXTURE0 + i, so each attachment gets its own "spot"
183  GLContextManager::getContext()->setActiveTexture(GL_TEXTURE0 + i);
184 
185  // Now bind the texture to its slot
186  GLContextManager::getContext()->bindTexture(GL_TEXTURE_2D, m_textureIDs[i]);
187 
188  // Just define the texture's basic properties - how it stores data, its size, etc. No data is added here.
189  GLenum attachmentType;
190  switch(i) {
191  case 0:
192  // Just a regular texture.
193  attachmentType = GL_COLOR_ATTACHMENT0;

200  break;
