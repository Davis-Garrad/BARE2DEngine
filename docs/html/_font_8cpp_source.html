246 glm::vec2 Font::measure(const char* s)
247 {
248 // Measure the text if it were being drawn with this font
249 // It is fixed height, so we just need to calculate the x value of the size, plus any newlines that contribute to height
250 glm::vec2 size(0, m_height);
251
252 float columnWidth = 0.0f;
253
254 // Loop through every character in the given text
255 for(int charIndex = 0; s[charIndex] != 0; charIndex++) {
256 char c = s[charIndex];
257 if(c == '\n') {
258 // We were given a new line, add to the height and (possibly) set the width of the whole text
259 size.y += m_height;
260 if(size.x < columnWidth) {
261 size.x = columnWidth;
262 }
263
264 // Reset the column width for a new column
265 columnWidth = 0;
266 } else {
267 // Check to make sure we actually have this glyph
268 unsigned int glyphIndex = c - m_regStart;
269 if(glyphIndex < 0 || glyphIndex >= m_regLength) {
270 // We don't actually have that character in the font, just add the width of the unknown characer
271 glyphIndex = m_regLength;
272 }
273 columnWidth += m_characterGlyphs[glyphIndex].size.x;
274 }
275 }
276
277 // One last time, just make sure that our column width is right
278 if(size.x < columnWidth)
279 size.x = columnWidth;
280
281 return size;
282 }

284 std::vector<int>* Font::createRows(glm::ivec4* rectangles, int rectanglesLength, int rows, int padding, int& width) {
285 // Create the returning vector
286 std::vector<int>* ret = new std::vector<int>[rows]();
287
288 // initialize some column widths. Initially, just the padding. Adding glyphs adds width obviously
289 int* columnWidths = new int[rows]();
290 // Set initial values.
291 for(int i = 0; i < rows; i++) {
292 columnWidths[i] = padding;
293 }
294
295 // Loop through every glyph, adding them as necessary
296 for(int i = 0; i < rectanglesLength; i++) {
297 // Find what row this should be in (the thinnest)
298 int rowIndex = 0;
299 for(int rowIndexTest = 0; rowIndexTest < rows; rowIndexTest++) {
300 // Test each row index, search for least wide
301 if(columnWidths[rowIndexTest] < columnWidths[rowIndex]) rowIndex = rowIndexTest;
302 }
303
304 // Add this glyph to that row (just add its width, and the other side of the padding)
305 columnWidths[rowIndex] += rectangles[i].z + padding;
306
307 // Actually add the index of the glyph added, for future reference
308 ret[rowIndex].push_back(i);
309 }
310
311 // Record the max width, which is put into `width`
312 width = 0;
313 for(int i = 0; i < rows; i++) {
314 if(columnWidths[i] > width) width = columnWidths[i];
315 }
316
317 delete[] columnWidths;
318
319 return ret;
320 }
